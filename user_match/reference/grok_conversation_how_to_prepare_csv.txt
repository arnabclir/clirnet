import pandas as pd
import hashlib

# Load cleaned CSV
df = pd.read_csv('cleaned_doctor_data.csv')

# Function to clean and hash a field
def clean_and_hash(value, field_type):
    if pd.isna(value):
        return ''
    if field_type == 'email':
        clean = str(value).lower().strip().replace(' ', '')
    elif field_type == 'phone':
        clean = '+' + str(value).replace(' ', '').replace('-', '').replace('(', '').replace(')', '')
    elif field_type in ['first_name', 'last_name']:
        clean = str(value).lower().strip().replace(' ', '')
    else:
        return value  # No hash for country/zip
    if field_type in ['email', 'phone', 'first_name', 'last_name']:
        hashed = hashlib.sha256(clean.encode('utf-8')).hexdigest()
        return hashed
    return clean

# Apply hashing
df['Email'] = df['Email'].apply(lambda x: clean_and_hash(x, 'email'))
df['Phone'] = df['Phone'].apply(lambda x: clean_and_hash(x, 'phone'))
df['First Name'] = df['First Name'].apply(lambda x: clean_and_hash(x, 'first_name'))
df['Last Name'] = df['Last Name'].apply(lambda x: clean_and_hash(x, 'last_name'))
# Country and Zip remain unhashed

# Save hashed CSV
df.to_csv('hashed_doctor_data.csv', index=False)
print("Hashing complete. Check hashed_doctor_data.csv")
